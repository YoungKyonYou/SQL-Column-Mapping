SELECT
DATE_FORMAT(co.A-COL1-TABLE1, '%Y-%m') AS order_month,
DATE_FORMAT(co.A-COL1-TABLE1, '%Y-%m-%d') AS order_day,
COUNT(DISTINCT co.A-COL2-TABLE1) AS num_orders,
COUNT(ol.A-COL3-TABLE2) AS num_books,
SUM(ol.A-COL4-TABLE2) AS total_price,
SUM(COUNT(ol.A-COL5-TABLE2)) OVER (
  PARTITION BY DATE_FORMAT(co.A-COL6-TABLE1, '%Y-%m')
  ORDER BY DATE_FORMAT(co.A-COL7-TABLE1, '%Y-%m-%d')
) AS running_total_num_books
FROM A-TB1 co
INNER JOIN A-TB2 ol ON co.A-COL8-TABLE1 = ol.A-COL9-TABLE2
GROUP BY 
  DATE_FORMAT(co.A-COL10-TABLE1, '%Y-%m'),
  DATE_FORMAT(co.A-COL10-TABLE1, '%Y-%m-%d')
ORDER BY co.A-COL10-TABLE1 ASC;
